  <script>
    const slotsByDate = {
      "2025-06-14": ["09:00 AM", "10:30 AM", "01:00 PM", "03:00 PM"],
      "2025-06-15": ["08:00 AM", "11:00 AM", "02:30 PM", "04:30 PM"],
      "2025-06-16": ["10:00 AM", "12:00 PM", "03:30 PM"]
    };

    let selectedSlot = null;

    function loadSlots() {
      const date = document.getElementById("dateInput").value;
      const container = document.getElementById("slotContainer");
      const msg = document.getElementById("confirmationMsg");

      container.innerHTML = "";
      msg.textContent = "";

      if (!date) {
        container.innerHTML = "<div class='no-slots'>Please select a date.</div>";
        return;
      }

      const slots = slotsByDate[date];

      if (!slots || slots.length === 0) {
        container.innerHTML = "<div class='no-slots'>No slots available for the selected date.</div>";
        return;
      }

      slots.forEach(time => {
        const btn = document.createElement("button");
        btn.className = "btn btn-outline-primary slot-button";
        btn.textContent = time;
        btn.onclick = () => {
          document.querySelectorAll('.slot-button').forEach(b => b.classList.remove('selected'));
          btn.classList.add("selected");
          selectedSlot = time;
          msg.textContent = `âœ… You selected: ${date} at ${time}`;
        };
        container.appendChild(btn);
      });
    }
  </script>
